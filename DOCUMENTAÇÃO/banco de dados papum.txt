CREATE DATABASE papum;

use papum;

create table tb_usuario(	
    id_usuario INT PRIMARY KEY AUTO_INCREMENT,	
    nome_usuario VARCHAR(100),	
    email_usuario VARCHAR(100),	
    senha_usuario VARCHAR(30), 
    cargo VARCHAR(20) NOT NULL DEFAULT 'estudante'
    
);

create table tb_tarefa(
	id_tarefa INT PRIMARY KEY AUTO_INCREMENT,
   	    nome_tarefa VARCHAR(50),
    	detalhamento_tarefa VARCHAR(500),
    	data_tarefa DATE,
    	prazo_tarefa DATE,
    	status_tarefa VARCHAR(50),
		usuario_tarefa INT,
	FOREIGN KEY  (usuario_tarefa) REFERENCES tb_usuario(id_usuario)
);

CREATE TABLE tb_evento(
	id_evento INT AUTO_INCREMENT PRIMARY KEY, 
    titulo_evento VARCHAR(100),
    data_evento DATE, 
    horario_evento TIME,
    data_prazo DATE,
    descricao VARCHAR(100),
    id_estudante INT,
    hora_prazo TIME,
    FOREIGN KEY tb_evento(id_estudante) REFERENCES tb_usuario(id_usuario)
);

CREATE TABLE tb_duvidas(
	id_duvidas INT PRIMARY KEY AUTO_INCREMENT,
	titulo_duvidas VARCHAR(60),
	texto_duvidas VARCHAR(500),
	data_duvidas date,
	usuario_duvidas INT,
	FOREIGN KEY (usuario_duvidas) REFERENCES tb_usuario(id_usuario)
